-- --- UPDATED PERFECT DRAGGING LOGIC WITH PC MOUSE SUPPORT ---
local function makeSmartDraggable(button, callback, isSmallButton)
    local dragging, dragInput, dragStart, startPos

    button.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
            if isSmallButton and buttonsLocked then
                local connection
                connection = input.Changed:Connect(function()
                    if input.UserInputState == Enum.UserInputState.End then
                        connection:Disconnect()
                        if callback then callback() end
                    end
                end)
                return
            end
            
            dragging = true
            dragStart = input.Position
            startPos = button.Position
            
            local connection
            connection = input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    if not dragging then return end
                    dragging = false
                    connection:Disconnect()
                    local delta = input.Position - dragStart
                    if delta.Magnitude < 6 then
                        if callback then callback() end
                    end
                end
            end)
        end
    end)

    button.InputChanged:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            button.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

-- Universal Chat Function (Fix for modern duplication)
local function speak(message)
    local sentModern = false
	if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
		local channel = TextChatService:FindFirstChild("TextChannels") and TextChatService.TextChannels:FindFirstChild("RBXGeneral")
		if channel then
			channel:SendAsync(message)
            sentModern = true
		end
	end
    if not sentModern then
        local sayMsg = ReplicatedStorage:FindFirstChild("SayMessageRequest", true)
        if sayMsg and sayMsg:IsA("RemoteEvent") then
            sayMsg:FireServer(message, "All")
        end
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Head") then
            Chat:Chat(LocalPlayer.Character.Head, message, Enum.ChatColor.White)
        end
    end
end

-- Fling Logic
local function startFlingLoop()
	if flingConnection then flingConnection:Disconnect() end
	flingConnection = RunService.Heartbeat:Connect(function()
		if not (flingEnabled or isSlapping) then return end
		local character = LocalPlayer.Character
		local hrp = character and character:FindFirstChild("HumanoidRootPart")
		if not hrp then return end
		task.spawn(function()
			local oldVelo = hrp.AssemblyLinearVelocity
			local oldRot = hrp.AssemblyAngularVelocity
			
			-- Applying extreme force
			hrp.AssemblyLinearVelocity = Vector3.new(45000000, 450000000, 45000000)
			hrp.AssemblyAngularVelocity = Vector3.new(99999, 99999, 99999)
			
			RunService.RenderStepped:Wait()
			
			-- Restoration
			if hrp and hrp.Parent then
				hrp.AssemblyLinearVelocity = oldVelo
				hrp.AssemblyAngularVelocity = oldRot
			end
		end)
	end)
end

local function stopFlingLoop()
	if flingConnection then
		flingConnection:Disconnect()
		flingConnection = nil
	end
end

local function spawnEarthquake(pos, color, material, intensityMult)
	task.spawn(function()
		local actualIntensity = intensityMult
		local baseCount = 15
		local count = math.floor(baseCount * actualIntensity)
		for i = 1, count do
			local angle = (i / count) * math.pi * 2
			local dist = math.random(4, 12) * actualIntensity
			local offset = Vector3.new(math.cos(angle) * dist, 0, math.sin(angle) * dist)
			local size = (math.random(100, 300) / 100) * actualIntensity
			local part = Instance.new("Part")
			part.Size = Vector3.new(size, size, size)
			part.Position = pos + offset + Vector3.new(0, -1, 0)
			part.Color = color
			part.Material = material
			part.CanCollide = false
			part.Anchored = false
			part.Parent = workspace
			local jumpPower = 30 * actualIntensity
			local velocity = (Vector3.new(0, jumpPower, 0) + (offset.Unit * (10 * actualIntensity)))
			part.AssemblyLinearVelocity = velocity
			part.AssemblyAngularVelocity = Vector3.new(math.random(-20, 20), math.random(-20, 20), math.random(-20, 20))
			Debris:AddItem(part, 2.5)
			task.spawn(function()
				task.wait(1.5)
				TweenService:Create(part, TweenInfo.new(1), {Transparency = 1}):Play()
			end)
		end
	end)
end

local function applyShake(intensity, duration)
	local camera = Workspace.CurrentCamera
	local startTime = tick()
	task.spawn(function()
		while tick() - startTime < duration do
			local shake = Vector3.new(
				math.random(-100, 100)/1000 * intensity,
				math.random(-100, 100)/1000 * intensity,
				math.random(-100, 100)/1000 * intensity
			)
			camera.CFrame = camera.CFrame * CFrame.new(shake)
			RunService.RenderStepped:Wait()
		end
	end)
end

-- --- UPDATED STAGE 3 SMOOTH SHAKE ENGINE ---
local function applyStage3Shake()
	local camera = Workspace.CurrentCamera
	task.spawn(function()
		task.wait(0.5)
		while mobileFlyOn and currentStage == 3 and not isLanding and not atkHolding and not isSlapping do
			local baseIntensity = 0.5
			local extraIntensity = (currentFlyMultiplier / 10) * 0.7
			targetStage3ShakeIntensity = baseIntensity + extraIntensity
			currentStage3ShakeIntensity = currentStage3ShakeIntensity + (targetStage3ShakeIntensity - currentStage3ShakeIntensity) * 0.25
			if currentStage3ShakeIntensity > 0.01 then
				local shake = Vector3.new(
					math.random(-100, 100)/1000 * currentStage3ShakeIntensity,
					math.random(-100, 100)/1000 * currentStage3ShakeIntensity,
					math.random(-100, 100)/1000 * currentStage3ShakeIntensity
				)
				camera.CFrame = camera.CFrame * CFrame.new(shake)
			end
			RunService.RenderStepped:Wait()
		end
	end)
end

local function applyAttackShake()
	local camera = Workspace.CurrentCamera
	task.spawn(function()
		local currentIntensity = 0
		local maxStage3Intensity = 0.5 + (10.0 / 10) * 0.7
		local targetMaxIntensity = (maxStage3Intensity * 2.0) / 1.5
		while atkHolding do
			local dt = RunService.RenderStepped:Wait()
			if attackIsMoving then
				currentIntensity = math.min(currentIntensity + (dt * 2), targetMaxIntensity)
			else
				currentIntensity = math.max(currentIntensity - (dt * 1.6), 0)
			end
			if currentIntensity > 0.01 then
				local shake = Vector3.new(
					math.random(-100, 100)/1000 * currentIntensity,
					math.random(-100, 100)/1000 * currentIntensity,
					math.random(-100, 100)/1000 * currentIntensity
				)
				camera.CFrame = camera.CFrame * CFrame.new(shake)
			end
		end
	end)
end

local function clearFlyTracks(fadeOverride)
	local useFade = fadeOverride or FADE_TIME
	for _, track in pairs(flyTracks) do track:Stop(useFade) end
	if flyAnimTask then local status = coroutine.status(flyAnimTask) if status ~= "dead" then task.cancel(flyAnimTask) end flyAnimTask = nil end
	if customFlyLoopTask then local status = coroutine.status(customFlyLoopTask) if status ~= "dead" then task.cancel(customFlyLoopTask) end customFlyLoopTask = nil end
	flyTracks = {}
end

local function playFlyAnim(id, looped, fadeOverride)
	-- GUARD: never play fly animations when fly is off
	if not mobileFlyOn then return nil end
	local character = LocalPlayer.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	if not humanoid then return end
	local useFade = fadeOverride or FADE_TIME
	local anim = Instance.new("Animation")
	anim.AnimationId = id
	local track = humanoid:LoadAnimation(anim)
	track.Looped = looped
	track:Play(useFade)
	table.insert(flyTracks, track)
	return track
end

-- Custom Ping-Pong Loop for Fly Animation V1, V2 & V3 Modifications
local function runCustomFlyLoop(track)
    if customFlyLoopTask then pcall(function() task.cancel(customFlyLoopTask) end) end
    customFlyLoopTask = task.spawn(function()
        if flyAnimVersion == 2 then
            track:Play()
            track:AdjustSpeed(0.5)
            while mobileFlyOn do
                RunService.RenderStepped:Wait()
            end
        elseif flyAnimVersion == 3 then
            track:Play()
            track:AdjustSpeed(1.0)
            while mobileFlyOn do
                RunService.RenderStepped:Wait()
            end
        else
            local START_TIME = 2.5
            local END_TIME = 2.7
            local PLAYBACK_SPEED = 0.2
            track:Play()
            while mobileFlyOn do
                track:AdjustSpeed(PLAYBACK_SPEED)
                track.TimePosition = START_TIME
                while track.TimePosition < END_TIME and mobileFlyOn do
                    RunService.RenderStepped:Wait()
                end
                if not mobileFlyOn then break end
                track:AdjustSpeed(-PLAYBACK_SPEED)
                track.TimePosition = END_TIME
                while track.TimePosition > START_TIME and mobileFlyOn do
                    RunService.RenderStepped:Wait()
                end
                if not mobileFlyOn then break end
            end
        end
        if not mobileFlyOn then
            pcall(function() track:Stop(0) end)
        end
    end)
end

-- Fly Core
updateFlyState = function(isMoving)
	-- GUARD: never run fly state logic when fly is off
	if not mobileFlyOn then return end
	if isLanding or isSlapping then return end
	attackIsMoving = isMoving
	if flyCurrentState == isMoving and not (foolToggled or thraxsaToggled or kingToggled) then return end
    if flyCurrentState ~= isMoving then
        flyCurrentState = isMoving
        if not isMoving then
            if not (foolToggled or thraxsaToggled or kingToggled) then
                clearFlyTracks(0.45)
                playFlyAnim(FLY_IDLE_OPTIONS[flyIdleVersion], true, 0.45)
            end
            currentFlyMultiplier = 1.0
            currentStage = 1
            stage3TimeElapsed = 0
        else
            if foolToggled or thraxsaToggled or kingToggled then
                if flyAnimTask then pcall(function() task.cancel(flyAnimTask) end) end
                flyAnimTask = task.spawn(function()
                    currentStage = 1
                    stage3TimeElapsed = 0
                    local startTime = tick()
                    while tick() - startTime < 0.173 and flyCurrentState == true and mobileFlyOn do
                        local alpha = (tick() - startTime) / 0.173
                        currentFlyMultiplier = 1.0 + (alpha * 1.0)
                        task.wait()
                    end
                    if flyCurrentState ~= true or not mobileFlyOn then return end
                    currentStage = 2
                    startTime = tick()
                    while tick() - startTime < 0.254 and flyCurrentState == true and mobileFlyOn do
                        local alpha = (tick() - startTime) / 0.254
                        currentFlyMultiplier = 2.0 + (alpha * 3.0)
                        task.wait()
                    end
                    if flyCurrentState ~= true or not mobileFlyOn then return end
                    currentStage = 3
                    applyStage3Shake()
                    local stage3StartTime = tick()
                    while flyCurrentState == true and mobileFlyOn do
                        local elapsed = tick() - stage3StartTime
                        stage3TimeElapsed = elapsed
                        local alpha = math.min(elapsed / 5, 1)
                        currentFlyMultiplier = 5.0 + (alpha * 5.0)
                        task.wait()
                    end
                end)
                return
            end
            if atkHolding then
                clearFlyTracks(0.5)
                if currentStage == 3 then currentFlyMultiplier = 10.0 end
                return
            end
            clearFlyTracks(0.5)
            flyAnimTask = task.spawn(function()
                if currentStage == 3 then
                    if not mobileFlyOn then return end
                    local activeTrackId = (flyAnimVersion == 3) and WALK_3_V3_ID or ((flyAnimVersion == 2) and WALK_3_V2_ID or WALK_3_ID)
                    local track = playFlyAnim(activeTrackId, (flyAnimVersion ~= 1), 0.5)
                    if track then
                        track.Priority = Enum.AnimationPriority.Action
                        runCustomFlyLoop(track)
                    end
                    applyStage3Shake()
                    return
                end
                currentStage = 1
                stage3TimeElapsed = 0
                local startTime = tick()
                local duration = 0.173
                local t1 = playFlyAnim(WALK_1_ID, false, 0.5)
                if t1 then t1:AdjustSpeed(2.6) end
                while tick() - startTime < duration and mobileFlyOn do
                    local alpha = (tick() - startTime) / duration
                    currentFlyMultiplier = 1.0 + (alpha * 1.0)
                    task.wait()
                end
                if not mobileFlyOn then return end
                if t1 then t1:Stop(0.5) end
                currentStage = 2
                startTime = tick()
                local duration = 0.254
                local t2 = playFlyAnim(WALK_2_ID, false, 0.5)
                if t2 then t2:AdjustSpeed(2.6) end
                while tick() - startTime < duration and mobileFlyOn do
                    local alpha = (tick() - startTime) / duration
                    currentFlyMultiplier = 2.0 + (alpha * 3.0)
                    task.wait()
                end
                if not mobileFlyOn then return end
                if t2 then t2:Stop(0.5) end
                currentStage = 3
                if not mobileFlyOn then return end
                local activeTrackId = (flyAnimVersion == 3) and WALK_3_V3_ID or ((flyAnimVersion == 2) and WALK_3_V2_ID or WALK_3_ID)
                local track = playFlyAnim(activeTrackId, (flyAnimVersion ~= 1), 0.5)
                if track then
                    track.Priority = Enum.AnimationPriority.Action
                    runCustomFlyLoop(track)
                end
                applyStage3Shake()
                local stage3StartTime = tick()
                local accelerationDuration = 5
                local stage3StartMult = 5.0
                local stage3MaxMult = 10.0
                while flyCurrentState == true and mobileFlyOn do
                    local elapsed = tick() - stage3StartTime
                    stage3TimeElapsed = elapsed
                    local alpha = math.min(elapsed / accelerationDuration, 1)
                    currentFlyMultiplier = stage3StartMult + (alpha * (stage3MaxMult - stage3StartMult))
                    task.wait()
                end
            end)
        end
    end
end

local function getFlyMovementDirection()
	local character = LocalPlayer.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local camera = Workspace.CurrentCamera
	if not humanoid or not camera then return Vector3.new() end
	local moveDirection = Vector3.new()
	local keyboardInputActive = false
	if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDirection = moveDirection + camera.CFrame.LookVector keyboardInputActive = true end
	if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDirection = moveDirection - camera.CFrame.LookVector keyboardInputActive = true end
	if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDirection = moveDirection - camera.CFrame.RightVector keyboardInputActive = true end
	if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDirection = moveDirection + camera.CFrame.RightVector keyboardInputActive = true end
	local verticalDir = Vector3.new(0, 0, 0)
	if UserInputService:IsKeyDown(Enum.KeyCode.Space) then local upBoost = (currentStage == 3) and 2.0 or 1.0; verticalDir = verticalDir + Vector3.new(0, 1 * upBoost, 0) keyboardInputActive = true end
	if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then local downBoost = (currentStage == 3) and 10.0 or 1.0; verticalDir = verticalDir - Vector3.new(0, 1 * downBoost, 0) keyboardInputActive = true end
	moveDirection = moveDirection + verticalDir
	if not keyboardInputActive and humanoid.MoveDirection.Magnitude > 0.05 then
		local joystickInput = humanoid.MoveDirection
		local cameraLookFlat = Vector3.new(camera.CFrame.LookVector.X, 0, camera.CFrame.LookVector.Z).Unit
		local cameraRightFlat = Vector3.new(camera.CFrame.RightVector.X, 0, camera.CFrame.RightVector.Z).Unit
		local forwardComponent = joystickInput:Dot(cameraLookFlat)
		local rightComponent = joystickInput:Dot(cameraRightFlat)
		moveDirection = moveDirection + (camera.CFrame.RightVector * rightComponent)
		moveDirection = moveDirection + (camera.CFrame.LookVector * forwardComponent)
	end
	return moveDirection.Magnitude > 0 and moveDirection.Unit or Vector3.new()
end

-- Landing sequence ID so we can invalidate stale coroutines
local landingSessionId = 0

-- REFINED FE GROUND LANDING
-- No anchoring. Uses tick-based loop for clean fly-off detection.
-- Hard-sets root.CFrame every Heartbeat tick to pin position server-visibly via physics replication.
local function handleLanding(hitPosition, hitInstance)
    if isLanding or landingCooldown or not groundLandingEnabled then return end
    if (currentStage >= 2 or atkHolding) then
        isLanding = true
        landingCooldown = true
        landingSessionId = landingSessionId + 1
        local mySession = landingSessionId
        descentStartHeight = nil
        clearFlyTracks(0.1)

        local character = LocalPlayer.Character
        local root = character and character:FindFirstChild("HumanoidRootPart")
        local humanoid = character and character:FindFirstChildOfClass("Humanoid")

        local groundColor = hitInstance and hitInstance.Color or Color3.fromRGB(100, 100, 100)
        local groundMaterial = hitInstance and hitInstance.Material or Enum.Material.Plastic
        local effectiveMult = atkHolding and (12.0 * ATTACK_FLY_SPEED_MULTIPLIER) or currentFlyMultiplier
        local intensity = (effectiveMult / 5)

        spawnEarthquake(hitPosition, groundColor, groundMaterial, intensity)
        applyShake(intensity * 1.5, 0.8)

        -- Zero out velocity immediately
        if flyBodyVelocity then flyBodyVelocity.velocity = Vector3.new(0, 0, 0) end
        if root then
            root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
            root.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
        end

        -- Compute target landing CFrame
        local targetLandCFrame
        if root and humanoid then
            local rootOffset = (root.Size.Y / 2) + humanoid.HipHeight
            local landPos = hitPosition + Vector3.new(0, rootOffset, 0)
            local lookDir = root.CFrame.LookVector
            local flatLook = Vector3.new(lookDir.X, 0, lookDir.Z)
            if flatLook.Magnitude < 0.01 then flatLook = Vector3.new(0, 0, -1) else flatLook = flatLook.Unit end
            targetLandCFrame = CFrame.lookAt(landPos, landPos + flatLook)
            -- Hard-set immediately (pins server-side via physics replication, no anchor needed)
            root.CFrame = targetLandCFrame
        end

        -- Abort if fly was turned off before we even play the anim
        if not mobileFlyOn or landingSessionId ~= mySession then
            if humanoid then humanoid.AutoRotate = true; humanoid.PlatformStand = false end
            isLanding = false
            task.delay(0.15, function() landingCooldown = false end)
            return
        end

        local landTrack = playFlyAnim("rbxassetid://79698004825744", false, 0.1)
        if landTrack then
            landTrack:AdjustSpeed(2)

            -- Tick-based wait loop instead of task.wait — checks fly-off every Heartbeat
            local animHalfDur = math.max(0, (landTrack.Length / 2) - 0.5)
            local waitStart = tick()
            -- Pin position every Heartbeat during the animation so others see it correctly
            local pinConnection = RunService.Heartbeat:Connect(function()
                if not mobileFlyOn or landingSessionId ~= mySession then return end
                if root and targetLandCFrame then
                    root.CFrame = targetLandCFrame
                    root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
                    root.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
                end
            end)

            -- Tick-based polling loop
            while tick() - waitStart < animHalfDur do
                RunService.Heartbeat:Wait()
                if not mobileFlyOn or landingSessionId ~= mySession then break end
            end
            pinConnection:Disconnect()

            -- Abort if fly was turned off during wait
            if not mobileFlyOn or landingSessionId ~= mySession then
                clearFlyTracks(0)
                if humanoid then humanoid.AutoRotate = true; humanoid.PlatformStand = false end
                isLanding = false
                task.delay(0.15, function() landingCooldown = false end)
                return
            end

            currentStage = 1

            -- Only play post-land animations while fly is still on
            if mobileFlyOn and landingSessionId == mySession then
                if foolToggled and humanoid then
                    local a = Instance.new("Animation"); a.AnimationId = FOOL_ANIM_ID
                    foolTrack = humanoid:LoadAnimation(a)
                    foolTrack.Priority = Enum.AnimationPriority.Action3
                    foolTrack.Looped = true
                    foolTrack:Play(0.5)
                elseif thraxsaToggled then
                    launchThraxsansLoop()
                elseif kingToggled and humanoid then
                    local a = Instance.new("Animation"); a.AnimationId = KING_ANIM_ID
                    kingTrack = humanoid:LoadAnimation(a)
                    kingTrack.Priority = Enum.AnimationPriority.Action3
                    customFlyLoopTask = task.spawn(function()
                        local START_TIME = 5.2
                        local END_TIME = 5.8
                        local PLAYBACK_SPEED = 0.3
                        kingTrack:Play()
                        while kingToggled and mobileFlyOn and not isLanding do
                            kingTrack:AdjustSpeed(PLAYBACK_SPEED)
                            kingTrack.TimePosition = START_TIME
                            while kingTrack.TimePosition < END_TIME and kingToggled and mobileFlyOn and not isLanding do RunService.RenderStepped:Wait() end
                            if not kingToggled or isLanding or not mobileFlyOn then break end
                            kingTrack:AdjustSpeed(-PLAYBACK_SPEED)
                            kingTrack.TimePosition = END_TIME
                            while kingTrack.TimePosition > START_TIME and kingToggled and mobileFlyOn and not isLanding do RunService.RenderStepped:Wait() end
                        end
                        if not mobileFlyOn then
                            pcall(function() if kingTrack then kingTrack:Stop(0) end end)
                        end
                    end)
                else
                    playFlyAnim(FLY_IDLE_OPTIONS[flyIdleVersion], true, 0.45)
                end
            end

            -- Short hold after post-land anim starts (tick-based)
            local holdStart = tick()
            while tick() - holdStart < 0.5 do
                RunService.Heartbeat:Wait()
                if not mobileFlyOn or landingSessionId ~= mySession then break end
            end

            -- Abort if fly was turned off during hold
            if not mobileFlyOn or landingSessionId ~= mySession then
                clearFlyTracks(0)
                if humanoid then humanoid.AutoRotate = true; humanoid.PlatformStand = false end
                isLanding = false
                task.delay(0.15, function() landingCooldown = false end)
                return
            end
        end

        isLanding = false
        flyCurrentState = nil
        task.delay(0.3, function() landingCooldown = false end)
    end
end

local function stopFly()
    -- Immediately abort any ongoing landing sequence so the player can walk right away
    landingSessionId = landingSessionId + 1
    if isLanding then
        isLanding = false
        landingCooldown = false
        local _c = LocalPlayer.Character
        local _h = _c and _c:FindFirstChildOfClass("Humanoid")
        if _h then _h.AutoRotate = true; _h.PlatformStand = false end
        clearFlyTracks(0)
    end
	if flyBodyGyro then flyBodyGyro:Destroy() end
	if flyBodyVelocity then flyBodyVelocity:Destroy() end
	if flyConnections.camPart then flyConnections.camPart:Destroy() end
	clearFlyTracks(0)
	if foolTrack then foolTrack:Stop(0) foolTrack = nil end
	if thraxsaTrack then thraxsaTrack:Stop(0) thraxsaTrack = nil end
	if kingTrack then kingTrack:Stop(0) kingTrack = nil end
	foolToggled = false; thraxsaToggled = false; kingToggled = false
	foolStroke.Color = Color3.fromRGB(60, 60, 60)
	thraxsaStroke.Color = Color3.fromRGB(60, 60, 60)
	kingStroke.Color = Color3.fromRGB(60, 60, 60)
	-- Stop choose death if active
	if chooseDeathToggled then
		chooseDeathToggled = false
		chooseDeathStroke.Color = Color3.fromRGB(60, 60, 60)
		if chooseDeathLoopTask then pcall(function() task.cancel(chooseDeathLoopTask) end) chooseDeathLoopTask = nil end
		if chooseDeathTrack then pcall(function() chooseDeathTrack:Stop(0) end) chooseDeathTrack = nil end
	end
	flyCurrentState = nil
	currentFlyMultiplier = 1.0
	currentStage = 1
	stage3TimeElapsed = 0
	totalFlightTimeElapsed = 0
	descentStartHeight = nil
	if mobileFlyInvisiblePart then mobileFlyInvisiblePart:Destroy() mobileFlyInvisiblePart = nil end
	local character = LocalPlayer.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local camera = Workspace.CurrentCamera
	if camera and humanoid then
		camera.CameraSubject = humanoid
		camera.CameraType = Enum.CameraType.Custom
		humanoid.AutoRotate = true
		humanoid.PlatformStand = false
		TweenService:Create(camera, TweenInfo.new(0.7, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {FieldOfView = 70}):Play()
	end
	if flyConnections.flyLoop then flyConnections.flyLoop:Disconnect(); flyConnections.flyLoop = nil end
	-- Final safety sweep: cancel any tasks and stop any tracks that may have slipped through
	task.delay(0.05, function()
		if not mobileFlyOn then
			clearFlyTracks(0)
			if foolTrack then pcall(function() foolTrack:Stop(0) end) foolTrack = nil end
			if thraxsaTrack then pcall(function() thraxsaTrack:Stop(0) end) thraxsaTrack = nil end
			if kingTrack then pcall(function() kingTrack:Stop(0) end) kingTrack = nil end
		end
	end)
end

		mobileFlyInvisiblePart = Instance.new("Part"); mobileFlyInvisiblePart.Name = "MobileFlyGroundPart"; mobileFlyInvisiblePart.Size = Vector3.new(10, 10, 10); mobileFlyInvisiblePart.Transparency = 1; mobileFlyInvisiblePart.CanCollide = false; mobileFlyInvisiblePart.Anchored = false; mobileFlyInvisiblePart.Massless = true; mobileFlyInvisiblePart.Parent = workspace
	end
	flyBodyGyro = Instance.new("BodyGyro"); flyBodyGyro.P = 40000; flyBodyGyro.D = 500; flyBodyGyro.maxTorque = Vector3.new(9e9, 9e9, 9e9); flyBodyGyro.Parent = root
	flyBodyVelocity = Instance.new("BodyVelocity"); flyBodyVelocity.velocity = Vector3.zero; flyBodyVelocity.maxForce = Vector3.new(9e9, 9e9, 9e9); flyBodyVelocity.Parent = root
	if head then
		local camPart = Instance.new("Part"); camPart.Name = "CamFocus"; camPart.Transparency = 1; camPart.CanCollide = false; camPart.Massless = true; camPart.Size = Vector3.new(0.1, 0.1, 0.1); camPart.Parent = character
		local weld = Instance.new("Weld"); weld.Part0 = head; weld.Part1 = camPart; weld.C0 = CFrame.new(0, 0.5, 0); weld.Parent = camPart
		camera.CameraSubject = camPart
		flyConnections.camPart = camPart
	end

	totalFlightTimeElapsed = 0
	flyConnections.flyLoop = RunService.RenderStepped:Connect(function(dt)
		if not mobileFlyOn or not root or not camera or isLanding or isSlapping then
			if currentStage3ShakeIntensity > 0 then
				currentStage3ShakeIntensity = math.max(currentStage3ShakeIntensity - (dt * 4), 0)
				if currentStage3ShakeIntensity > 0 then
					local shake = Vector3.new(
						math.random(-100, 100)/1000 * currentStage3ShakeIntensity,
						math.random(-100, 100)/1000 * currentStage3ShakeIntensity,
						math.random(-100, 100)/1000 * currentStage3ShakeIntensity
					)
					camera.CFrame = camera.CFrame * CFrame.new(shake)
				end
			end
			return
		end

		totalFlightTimeElapsed = totalFlightTimeElapsed + dt

		if currentStage ~= 3 or not flyCurrentState then
			if currentStage3ShakeIntensity > 0 then
				currentStage3ShakeIntensity = math.max(currentStage3ShakeIntensity - (dt * 4), 0)
				local shake = Vector3.new(
					math.random(-100, 100)/1000 * currentStage3ShakeIntensity,
					math.random(-100, 100)/1000 * currentStage3ShakeIntensity,
					math.random(-100, 100)/1000 * currentStage3ShakeIntensity
				)
				camera.CFrame = camera.CFrame * CFrame.new(shake)
			end
		end

		local moveDir = getFlyMovementDirection()
		local filterList = {character, mobileFlyInvisiblePart}
		for _, p in ipairs(Players:GetPlayers()) do
			if p.Character then table.insert(filterList, p.Character) end
		end
		local rayCastParams = RaycastParams.new(); rayCastParams.FilterDescendantsInstances = filterList; rayCastParams.FilterType = Enum.RaycastFilterType.Exclude

		local currentVelY = flyBodyVelocity and flyBodyVelocity.velocity.Y or root.AssemblyLinearVelocity.Y
		local isMovingDown = currentVelY < -5

		if isMovingDown then
			if descentStartHeight == nil then
				local longRay = workspace:Raycast(root.Position, Vector3.new(0, -500, 0), rayCastParams)
				descentStartHeight = longRay and (root.Position.Y - longRay.Position.Y) or 999
			end
		else
			descentStartHeight = nil
		end

		local dynamicDetectionDist = math.max(3, ((flyBodyVelocity and flyBodyVelocity.velocity.Magnitude or root.AssemblyLinearVelocity.Magnitude) * dt * 0.75) + 1)
		local groundRay = workspace:Raycast(root.Position, Vector3.new(0, -dynamicDetectionDist, 0), rayCastParams)

		if groundRay and groundRay.Instance and groundLandingEnabled and isMovingDown then
			if groundRay.Instance.CanCollide and groundRay.Instance.Transparency <= 0.75 and (currentStage >= 2 or atkHolding) then
				local currentSpeed = flyBodyVelocity and flyBodyVelocity.velocity.Magnitude or root.AssemblyLinearVelocity.Magnitude
				if currentSpeed > 30 and (descentStartHeight or 0) >= MIN_DESCENT_FOR_LANDING then
					handleLanding(groundRay.Position, groundRay.Instance)
				end
			end
		end

		if moveDir.Magnitude > 0 then
			local speed = (baseFlySpeed * currentFlyMultiplier) * FLY_SPEED_MODIFIERS[flySpeedMode]
			if atkHolding then speed = (baseFlySpeed * 12.0) * FLY_SPEED_MODIFIERS[flySpeedMode] if attackStrikeInProgress then speed = speed / 4 end end
			flyBodyVelocity.velocity = moveDir * speed
			-- SHIFTLOCK / ROTATION TRACKING
			-- shiftlockEnabled: body faces camera direction including full pitch (look up/down freely)
			-- normal: body faces movement direction (cameraFollowOn) or flat movement direction
			if shiftlockEnabled then
				-- Full camera-aligned rotation including vertical look
				local targetRotation = CFrame.lookAt(root.Position, root.Position + camera.CFrame.LookVector)
				flyBodyGyro.CFrame = flyBodyGyro.CFrame:Lerp(targetRotation, 0.15)
			elseif cameraFollowOn then
				local targetRotation = CFrame.lookAt(root.Position, root.Position + moveDir)
				flyBodyGyro.CFrame = flyBodyGyro.CFrame:Lerp(targetRotation, 0.15)
			else
				local flatMove = Vector3.new(moveDir.X, 0, moveDir.Z)
				if flatMove.Magnitude > 0.05 then
					local targetFlatRotation = CFrame.lookAt(root.Position, root.Position + flatMove.Unit)
					flyBodyGyro.CFrame = flyBodyGyro.CFrame:Lerp(targetFlatRotation, 0.15)
				end
			end
			updateFlyState(true)
			if (currentStage == 3 and not (foolToggled or thraxsaToggled or kingToggled)) or atkHolding or (currentStage == 3 and (foolToggled or thraxsaToggled or kingToggled)) then
				if camera.FieldOfView < 89.9 then TweenService:Create(camera, TweenInfo.new(0.35), {FieldOfView = 90}):Play() end
			else
				if camera.FieldOfView > 70.1 then TweenService:Create(camera, TweenInfo.new(0.7), {FieldOfView = 70}):Play() end
			end
		else
			flyBodyVelocity.velocity = Vector3.zero
			-- SHIFTLOCK IDLE: always face camera direction including pitch when idle too
			if shiftlockEnabled then
				local targetRotation = CFrame.lookAt(root.Position, root.Position + camera.CFrame.LookVector)
				flyBodyGyro.CFrame = flyBodyGyro.CFrame:Lerp(targetRotation, 0.1)
			elseif cameraFollowOn then
				local targetIdleRotation = CFrame.lookAt(root.Position, root.Position + camera.CFrame.LookVector)
				flyBodyGyro.CFrame = flyBodyGyro.CFrame:Lerp(targetIdleRotation, 0.1)
			end
			updateFlyState(false)
			if camera.FieldOfView ~= 70 then TweenService:Create(camera, TweenInfo.new(0.7), {FieldOfView = 70}):Play() end
		end
		pcall(function()
			if mobileFlyInvisiblePart and root then
				local verticalIdleAdjustment = (moveDir.Magnitude == 0) and -6.5 or -5
				mobileFlyInvisiblePart.CFrame = root.CFrame * CFrame.new(0, -(root.Size.Y / 2) + verticalIdleAdjustment, 0)
			end
		end)
	end)
	updateFlyState(getFlyMovementDirection().Magnitude > 0)
end
